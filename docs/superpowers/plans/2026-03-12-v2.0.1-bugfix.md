comparison = industry_pos.get("comparison_table", [])
if comparison:
    lines.append("")
    lines.append("| 公司 | PE(TTM) | PB | 定位 |")
    lines.append("|:-----|:--------|:---|:-----|")
    for row in comparison:
        name = row.get("name", row.get("ticker", "?"))

        pb = f"{row['pb']:.1f}x" if row.get("pb") else "—"

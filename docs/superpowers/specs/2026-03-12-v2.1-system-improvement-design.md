```python
    """安全格式化，处理None值"""
    if value is None:
```

### 5.1 Task 3.1: 消费护城河公司估值修正

**Bug ID**: BUG-06
**影响**: 茅台Graham权重41.7%，分众目标价¥3.30与基本面80分矛盾
**文件**: `src/data/industry_mapping.py`, `src/agents/valuation.py`

```python
# 护城河检测标准
BRAND_MOAT_CRITERIA = {
    'gross_margin_min': 0.60,      # 毛利率 >= 60%
    'roe_5yr_avg_min': 0.15,       # 5年平均ROE >= 15%
    'fcf_positive_years_min': 4,   # 至少4年正向FCF
    'revenue_growth_stable': True,  # 营收稳定增长
}

def detect_brand_moat(metrics: dict) -> bool:
    """
    检测公司是否具有消费护城河特征
    满足3个以上条件即判定为护城河型
    """
    conditions_met = 0

    if metrics.get('gross_margin', 0) >= BRAND_MOAT_CRITERIA['gross_margin_min']:
        conditions_met += 1

    if metrics.get('roe_5yr_avg', 0) >= BRAND_MOAT_CRITERIA['roe_5yr_avg_min']:
        conditions_met += 1

    fcf_history = metrics.get('fcf_history', [])
    positive_years = sum(1 for fcf in fcf_history if fcf > 0)
    if positive_years >= BRAND_MOAT_CRITERIA['fcf_positive_years_min']:
        conditions_met += 1

    revenue_growth = metrics.get('revenue_growth_5yr', [])
    if len(revenue_growth) >= 3 and all(g > 0 for g in revenue_growth[-3:]):
        conditions_met += 1

    return conditions_met >= 3

def calculate_pb_from_roe(roe: float, cost_of_equity: float = 0.08) -> float:
    """
    基于ROE推导合理P/B倍数
    公式: P/B = ROE / Ke

    参数:
    - roe: 股本回报率 (如0.30表示30%)
    - cost_of_equity: 股权成本，默认8% (A股市场合理假设)

    示例:
    - 茅台: ROE=30%, Ke=8% → 合理P/B = 0.30/0.08 = 3.75x
    - 市场实际给6-8x，反映市场对ROE持续性的预期溢价
    """
    if roe <= 0 or cost_of_equity <= 0:
        return 1.0  # 异常情况返回1倍P/B
    return roe / cost_of_equity

# 在valuation.py中集成护城河估值
def apply_brand_moat_valuation(metrics: dict, industry_config: dict) -> dict:
    """
    护城河型公司估值流程:
    1. 检测是否为护城河型
    2. 如果是，禁用Graham，使用P/B_ROE推导
    3. 计算基于ROE的合理P/B
    """
    if detect_brand_moat(metrics):
        # 更新行业配置为护城河型
        industry_config['industry_type'] = 'brand_moat'
        industry_config['disable_methods'] = ['graham_number']

        # 计算ROE推导的P/B
        roe = metrics.get('roe', 0.15)
        fair_pb = calculate_pb_from_roe(roe)
        bvps = metrics.get('bvps', 0)

        return {
            'method': 'pb_roe_derived',
            'fair_pb': fair_pb,
            'target_price': bvps * fair_pb,
            'note': f'护城河型公司，P/B基于ROE={roe:.1%}推导'
        }

    return None
```

```python
def evaluate_fcf(fcf, net_income, capex, revenue_growth, industry_type):
    # 投资型负FCF：高增长 + 高Capex → 不扣分
    if fcf < 0 and revenue_growth > 20 and capex > net_income * 0.5:
        return {
            'score_impact': 0,
            'fcf_type': 'growth_investment',
            'note': f'成长投资期：营收增速{revenue_growth:.1f}%'
        }

    # 经营型负FCF → 扣分
    if fcf < 0 and revenue_growth < 10:
        return {
            'score_impact': -15,
            'fcf_type': 'operational_issue'
        }
```

```python
    """

    A股退市规则参考:
    1. 连续两年亏损 → *ST
    2. 连续三年亏损 → 暂停上市
    3. 净资产为负 → 退市风险警示
    4. 审计意见非标准 → 关注
    """
    risk_factors = []
    risk_level = 'LOW'

    # 连续亏损检查
    net_income_history = metrics.get('net_income_history', [])
    consecutive_losses = 0
    for ni in reversed(net_income_history):
        if ni < 0:
            consecutive_losses += 1
        else:
            break

    if consecutive_losses >= 3:
        risk_factors.append('连续三年亏损')
        risk_level = 'HIGH'
    elif consecutive_losses >= 2:
        risk_factors.append('连续两年亏损')
        risk_level = 'MEDIUM' if risk_level == 'LOW' else risk_level

    # 净资产检查
    net_assets = metrics.get('net_assets', 0)
    if net_assets <= 0:
        risk_factors.append('净资产为负')
        risk_level = 'HIGH'
    elif net_assets < metrics.get('total_assets', 1) * 0.1:
        risk_factors.append('净资产占比过低')
        risk_level = 'MEDIUM' if risk_level == 'LOW' else risk_level

    # 审计意见检查
    audit_opinion = metrics.get('audit_opinion', '标准无保留')
    if audit_opinion in ['无法表示意见', '否定意见']:
        risk_factors.append(f'审计意见: {audit_opinion}')
        risk_level = 'HIGH'
    elif audit_opinion in ['保留意见', '带强调事项段']:
        risk_factors.append(f'审计意见: {audit_opinion}')
        risk_level = 'MEDIUM' if risk_level == 'LOW' else risk_level

    return {
        'level': risk_level,
        'factors': risk_factors,
        'consecutive_losses': consecutive_losses
    }

def distressed_valuation(metrics: dict) -> dict:
    """
    困境企业估值框架
    当传统估值方法不适用时，采用替代分析
    """
    results = {}

    # 1. EV/Sales（适用于亏损但有营收的企业）
    if metrics.get('revenue', 0) > 0:
        # 困境企业使用0.3x折价倍数
        ev = metrics['revenue'] * 0.3
        market_cap = metrics.get('market_cap', 0)
        debt = metrics.get('total_debt', 0)
        cash = metrics.get('cash', 0)
        implied_equity = ev - debt + cash
        results['ev_sales'] = {
            'value': implied_equity / metrics.get('shares', 1),
            'multiple': 0.3,
            'note': '困境折价EV/Sales'
        }

    # 2. Net-Net分析（格雷厄姆清算价值）
    current_assets = metrics.get('current_assets', 0)
    total_liabilities = metrics.get('total_liabilities', 0)
    if current_assets > total_liabilities:
        net_net_value = (current_assets - total_liabilities) / metrics.get('shares', 1)
        results['net_net'] = {
            'value': net_net_value,
            'current_assets': current_assets,
            'total_liabilities': total_liabilities,
            'note': f'流动资产{current_assets/1e8:.1f}亿 > 总负债{total_liabilities/1e8:.1f}亿'
        }

    # 3. 退市风险评估
    delisting_risk = is_delisting_risk(metrics)
    if delisting_risk['level'] in ['MEDIUM', 'HIGH']:
        results['delisting_risk'] = delisting_risk

    return results
```

```python
def fetch_eps(stock_code: str, fiscal_year: int) -> dict:
    """
    统一EPS获取口径

    优先级:
    1. 基本EPS（与市场报价一致，推荐）
    2. 全面摊薄EPS
    3. 计算值（净利润/总股本）

    AKShare字段映射:
    - 基本EPS: stock_financial_abstract_em -> '基本每股收益'
    - 全面摊薄EPS: stock_financial_analysis_indicator -> '摊薄每股收益(元)'
    - 计算用数据: 净利润/总股本
    """
    import akshare as ak

    result = {'eps': None, 'source': None, 'reliable': False}

    # 尝试1: 基本EPS（推荐，与市场报价一致）
    try:
        abstract_df = ak.stock_financial_abstract_em(symbol=stock_code)
        if not abstract_df.empty:
            # 找到对应年份的数据
            year_mask = abstract_df['报告期'].str.contains(str(fiscal_year))
            if year_mask.any():
                basic_eps = abstract_df.loc[year_mask, '基本每股收益'].iloc[0]
                if basic_eps is not None and not pd.isna(basic_eps):
                    result = {
                        'eps': float(basic_eps),
                        'source': 'basic_eps',
                        'reliable': True,
                        'field': '基本每股收益'
                    }
                    return result
    except Exception as e:
        logger.debug(f"获取基本EPS失败: {e}")

    # 尝试2: 全面摊薄EPS
    try:
        indicator_df = ak.stock_financial_analysis_indicator(symbol=stock_code)
        if not indicator_df.empty:
            year_mask = indicator_df['报告期'].str.contains(str(fiscal_year))
            if year_mask.any():
                diluted_eps = indicator_df.loc[year_mask, '摊薄每股收益(元)'].iloc[0]
                if diluted_eps is not None and not pd.isna(diluted_eps):
                    result = {
                        'eps': float(diluted_eps),
                        'source': 'diluted_eps',
                        'reliable': True,
                        'field': '摊薄每股收益(元)'
                    }
                    return result
    except Exception as e:
        logger.debug(f"获取摊薄EPS失败: {e}")

    # 尝试3: 计算值（最后手段）
    try:
        net_income = fetch_net_income(stock_code, fiscal_year)
        total_shares = fetch_total_shares(stock_code)

        if net_income is not None and total_shares is not None and total_shares > 0:
            calculated_eps = net_income / total_shares
            result = {
                'eps': calculated_eps,
                'source': 'calculated',

                'field': f'净利润({net_income/1e8:.2f}亿)/总股本({total_shares/1e8:.2f}亿股)',
                'warning': 'EPS为计算值，可能与报告值存在差异'
            }
            return result
    except Exception as e:
        logger.warning(f"计算EPS失败: {e}")

    return result

def validate_eps_consistency(eps_reported: float, eps_calculated: float,
                             threshold: float = 0.1) -> dict:
    """
    验证EPS一致性

    参数:
    - threshold: 允许的偏差阈值，默认10%

    返回:
    - is_consistent: 是否一致
    - deviation: 偏差百分比
    - recommendation: 建议使用哪个值
    """
    if eps_reported is None or eps_calculated is None:
        return {'is_consistent': True, 'deviation': None, 'recommendation': 'use_available'}

    if eps_reported == 0:
        return {'is_consistent': False, 'deviation': float('inf'), 'recommendation': 'use_calculated'}

    deviation = abs(eps_reported - eps_calculated) / abs(eps_reported)

    return {
        'is_consistent': deviation <= threshold,
        'deviation': deviation,
        'recommendation': 'use_reported' if deviation <= threshold else 'investigate'
    }
```

```python
def detect_data_contradictions(metrics: dict) -> list:
    """
    检测财务数据内部矛盾
    返回矛盾列表，每个矛盾包含类型、严重程度和详情
    """
    contradictions = []

    # 矛盾1：净利率与EPS不匹配
    net_margin = metrics.get('net_margin')
    eps = metrics.get('eps')
    revenue = metrics.get('revenue')
    shares = metrics.get('shares')

    if all(v is not None for v in [net_margin, eps, revenue, shares]) and shares > 0:
        implied_net_income = revenue * net_margin / 100  # 净利率隐含的净利润
        reported_net_income = eps * shares               # EPS隐含的净利润

        if implied_net_income > 0 and reported_net_income > 0:
            deviation = abs(implied_net_income - reported_net_income) / max(implied_net_income, 1)
            if deviation > 0.5:  # 偏差超过50%
                contradictions.append({
                    'type': 'net_margin_eps_mismatch',
                    'severity': 'high',
                    'detail': f"净利率{net_margin:.1f}%隐含净利{implied_net_income/1e8:.1f}亿，"
                             f"但EPS隐含净利{reported_net_income/1e8:.1f}亿，偏差{deviation:.0%}"
                })

    # 矛盾2：ROE历史跳变
    roe_current = metrics.get('roe')
    roe_5yr_avg = metrics.get('roe_5yr_avg')

    if roe_current is not None and roe_5yr_avg is not None:
        if abs(roe_current - roe_5yr_avg) > 20:  # ROE变化超过20个百分点
            contradictions.append({
                'type': 'roe_historical_jump',
                'severity': 'medium',
                'detail': f"当年ROE={roe_current:.1f}%与5年均值{roe_5yr_avg:.1f}%差异过大"
            })

    # 矛盾3：正净利但负OCF（连续两年）
    net_income = metrics.get('net_income')
    ocf = metrics.get('ocf')
    ocf_prev = metrics.get('ocf_prev_year')

    if net_income is not None and net_income > 0:
        if ocf is not None and ocf < 0:
            if ocf_prev is not None and ocf_prev < 0:
                contradictions.append({
                    'type': 'ni_ocf_divergence_persistent',
                    'severity': 'high',
                    'detail': f"净利润{net_income/1e8:.1f}亿为正，但经营现金流连续两年为负"
                             f"（今年{ocf/1e8:.1f}亿，去年{ocf_prev/1e8:.1f}亿）"
                })
            else:
                contradictions.append({
                    'type': 'ni_ocf_divergence',
                    'severity': 'medium',
                    'detail': f"净利润{net_income/1e8:.1f}亿为正，但经营现金流{ocf/1e8:.1f}亿为负"
                })

    # 矛盾4：EPS与报告EPS不一致
    eps_calculated = metrics.get('eps_calculated')
    eps_reported = metrics.get('eps_reported')

    if eps_calculated is not None and eps_reported is not None and eps_reported != 0:
        eps_deviation = abs(eps_calculated - eps_reported) / abs(eps_reported)
        if eps_deviation > 0.2:  # 偏差超过20%
            contradictions.append({
                'type': 'eps_inconsistency',
                'severity': 'low',
                'detail': f"计算EPS={eps_calculated:.2f}与报告EPS={eps_reported:.2f}不一致，偏差{eps_deviation:.0%}"
            })

    return contradictions

def get_data_confidence_score(contradictions: list) -> float:
    """
    基于矛盾计算数据可信度分数

    严重程度权重:
    - high: 扣0.3分
    - medium: 扣0.15分
    - low: 扣0.05分

    返回: 0.0-1.0的可信度分数
    """
    severity_weights = {'high': 0.3, 'medium': 0.15, 'low': 0.05}
    penalty = sum(severity_weights.get(c['severity'], 0.1) for c in contradictions)
    return max(0.0, 1.0 - penalty)

def generate_data_warning(contradictions: list, confidence: float) -> str:
    """
    当可信度<0.5时，生成报告开头的警告文本
    """
    if confidence >= 0.5:
        return None

    warning = f"""## ⚠️ 数据可信度警告

本报告使用的财务数据存在以下内部矛盾，可信度评分：**{confidence:.0%}**

| 矛盾类型 | 严重程度 | 详情 |
|:---------|:---------|:-----|
"""
    severity_map = {'high': '高', 'medium': '中', 'low': '低'}
    type_map = {
        'net_margin_eps_mismatch': '净利率/EPS不匹配',
        'roe_historical_jump': 'ROE历史跳变',
        'ni_ocf_divergence': '净利/OCF背离',
        'ni_ocf_divergence_persistent': '净利/OCF持续背离',
        'eps_inconsistency': 'EPS数据不一致'
    }

    for c in contradictions:
        type_name = type_map.get(c['type'], c['type'])
        severity = severity_map.get(c['severity'], c['severity'])
        warning += f"| {type_name} | {severity} | {c['detail']} |\n"

    warning += "\n**建议**：以下分析结论仅供参考，请核实原始财报数据后再做决策。\n"
    return warning
```

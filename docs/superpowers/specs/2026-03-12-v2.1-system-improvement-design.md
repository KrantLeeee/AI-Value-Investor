基于39份A股报告的批量测试，发现以下核心问题：

1. **防崩溃**：消除100%的NoneType报错

```python
def safe_format(value, fmt="{}", default="N/A"):
    """安全格式化，处理None值"""
    if value is None:
        return default
    try:
        return fmt.format(value)
    except (ValueError, TypeError):
        return default

# 应用到所有财务字段
roe_str = safe_format(metrics.get('roe'), "{:.1f}%")
net_margin_str = safe_format(metrics.get('net_margin'), "{:.1f}%")
```

```python
def calculate_ev_ebitda_value(ebitda, multiple, shares, revenue):
    if ebitda is None or ebitda <= 0:
        return None, "EBITDA数据无效，该方法不适用"

    ev = ebitda * multiple
    min_reasonable_ev = revenue * 0.1
    if ev < min_reasonable_ev:
        return None, "EV/EBITDA计算值异常，已排除"

    return ev / shares, None
```

### 5.2 Task 3.2: 成长期负FCF区分

**文件**: `src/agents/fundamentals.py`

```python
def evaluate_fcf(fcf, net_income, capex, revenue_growth, industry_type):
    # 投资型负FCF：高增长 + 高Capex → 不扣分
    if fcf < 0 and revenue_growth > 20 and capex > net_income * 0.5:
        return {
            'score_impact': 0,
            'fcf_type': 'growth_investment',
            'note': f'成长投资期：营收增速{revenue_growth:.1f}%'
        }

    # 经营型负FCF → 扣分
    if fcf < 0 and revenue_growth < 10:
        return {
            'score_impact': -15,
            'fcf_type': 'operational_issue'
        }
```

```python
def detect_distressed_company(metrics: dict) -> bool:
    signals = [
        metrics.get('net_margin', 0) < -20,
        metrics.get('roe', 0) < -15,
        metrics.get('fcf', 0) < 0 and metrics.get('ocf', 0) < 0,
        metrics.get('debt_equity', 0) > 300,
    ]
    return sum(signals) >= 2

def distressed_valuation(metrics: dict) -> dict:
    results = {}

    # EV/Sales
    if metrics.get('revenue', 0) > 0:
        results['ev_sales'] = calculate_ev_sales(metrics, multiple=0.3)

    # Net-Net分析
    current_assets = metrics.get('current_assets', 0)
    total_liabilities = metrics.get('total_liabilities', 0)
    if current_assets > total_liabilities:
        results['net_net'] = (current_assets - total_liabilities) / metrics['shares']

    # 退市风险评估
    if is_delisting_risk(metrics):
        results['delisting_risk'] = {'level': 'HIGH'}

    return results
```

```python
def detect_data_contradictions(metrics: dict) -> list:
    contradictions = []

    # 净利率与EPS不匹配
    # ROE历史跳变
    # 正净利但负OCF

    return contradictions

def get_data_confidence_score(contradictions: list) -> float:
    severity_weights = {'high': 0.3, 'medium': 0.15, 'low': 0.05}
    penalty = sum(severity_weights.get(c['severity'], 0.1) for c in contradictions)
    return max(0.0, 1.0 - penalty)
```

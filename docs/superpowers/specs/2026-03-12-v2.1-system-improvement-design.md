```python
def safe_format(value, fmt="{}", default="N/A"):
    """安全格式化，处理None值"""
    if value is None:
        return default
    try:
        return fmt.format(value)
    except (ValueError, TypeError):
        return default

# 应用到所有财务字段
roe_str = safe_format(metrics.get('roe'), "{:.1f}%")
```

```python
def calculate_ev_ebitda_value(ebitda, multiple, shares, revenue):
    if ebitda is None or ebitda <= 0:
        return None, "EBITDA数据无效，该方法不适用"

    ev = ebitda * multiple
    min_reasonable_ev = revenue * 0.1
```

```python
def match_keywords(company_name: str, business_desc: str,
                   akshare_industry: str) -> tuple:
    """
    关键词匹配，返回(行业类型, 置信度增量)

    匹配优先级：
    1. 公司名称中的primary关键词 → +0.3
    2. 业务描述中的primary关键词 → +0.25
    3. 公司名称中的secondary关键词 → +0.2
    4. AKShare行业字段匹配 → +0.15
    """
    combined_text = company_name + business_desc

    best_match = None
    best_score = 0.0

    for industry_type, keywords in INDUSTRY_KEYWORDS.items():
        # 检查排除词
        if any(neg in combined_text for neg in keywords.get('negative', [])):
            continue

        score = 0.0
        # Primary关键词匹配
        if any(kw in company_name for kw in keywords['primary']):
            score = 0.3
        elif any(kw in business_desc for kw in keywords['primary']):
            score = 0.25
        elif any(kw in company_name for kw in keywords.get('secondary', [])):
            score = 0.2

        if score > best_score:
            best_score = score
            best_match = industry_type

    return best_match, best_score

        'gross_margin': {'min': 15.0, 'max': 35.0},
        'customer_concentration': 'high',              # 客户集中度高
    },
}

def match_financial_profile(metrics: dict, industry_type: str) -> float:
    """
    检查财务特征是否符合行业画像
    返回: 0.0-0.2 的置信度增量
    """
    if industry_type not in FINANCIAL_PROFILES:
        return 0.0

    profile = FINANCIAL_PROFILES[industry_type]
    matches = 0
    total_checks = 0

    for metric_name, condition in profile.items():
        if isinstance(condition, dict):
            value = metrics.get(metric_name)
            if value is not None:
                total_checks += 1
                min_val = condition.get('min', float('-inf'))
                max_val = condition.get('max', float('inf'))
                if min_val <= value <= max_val:
                    matches += 1
        elif isinstance(condition, bool):
            # 布尔特征检查
            total_checks += 1
            if metrics.get(metric_name) == condition:
                matches += 1

    if total_checks == 0:
        return 0.0

    match_ratio = matches / total_checks
    return match_ratio * 0.2  # 最高+0.2

# ============================================================
# 置信度因素4：同业公司相似度 → +0.1
# ============================================================

def calculate_peer_similarity(stock_code: str, candidate_industry: str,
                              peer_list: list) -> float:
    """
    检查候选行业的代表公司列表中是否有相似公司

    相似度定义：
    - 同一申万二级行业 → +0.1
    - 业务描述关键词重叠率>50% → +0.05
    """
    # 从config/industry_profiles.yaml获取该行业的代表公司
    representative_companies = get_representative_companies(candidate_industry)

    # 检查是否在代表公司列表中
    if stock_code in representative_companies:
        return 0.1

    # 检查申万行业分类是否一致
    stock_sw_industry = get_shenwan_industry(stock_code)
    peer_sw_industries = [get_shenwan_industry(p) for p in representative_companies]

    if stock_sw_industry in peer_sw_industries:
        return 0.08

    return 0.0

# 主函数：整合所有置信度因素
# ============================================================

def classify_industry(stock_code: str, company_info: dict,
                      metrics: dict) -> IndustryClassificationResult:
    """
    多信号行业识别，返回带置信度的分类结果

    置信度计算:
    1. 精确匹配override列表 → +0.4
    2. 行业关键词匹配 → +0.2-0.3
    3. 财务特征符合行业画像 → +0.2
    4. 同业公司相似度 → +0.1

    总分最高1.0，阈值0.6以下触发保守模式
    """
    confidence = 0.0
    confidence_factors = {}
    classification_path = []

    # 因素1：Override检查
    override_result, override_score = check_override(stock_code)
    if override_result:
        confidence += override_score
        confidence_factors['override'] = override_score
        classification_path.append(f"Override匹配: {override_result}")
        # Override直接返回，不继续其他检查
        return IndustryClassificationResult(
            industry_type=override_result,
            display_name=get_display_name(override_result),
            confidence=min(confidence, 1.0),
            confidence_factors=confidence_factors,
            conservative_mode=False,
            classification_path=classification_path
        )

    # 因素2：关键词匹配
    company_name = company_info.get('name', '')
    business_desc = company_info.get('business_description', '')
    akshare_industry = company_info.get('akshare_industry', '')

    keyword_industry, keyword_score = match_keywords(
        company_name, business_desc, akshare_industry
    )
    if keyword_industry:
        confidence += keyword_score
        confidence_factors['keyword'] = keyword_score
        classification_path.append(f"关键词匹配: {keyword_industry} (+{keyword_score:.2f})")

    # 因素3：财务特征画像
    if keyword_industry:
        profile_score = match_financial_profile(metrics, keyword_industry)
        confidence += profile_score
        confidence_factors['financial_profile'] = profile_score
        classification_path.append(f"财务画像匹配: +{profile_score:.2f}")

    # 因素4：同业相似度
    if keyword_industry:
        peer_score = calculate_peer_similarity(stock_code, keyword_industry, [])
        confidence += peer_score
        confidence_factors['peer_similarity'] = peer_score
        classification_path.append(f"同业相似度: +{peer_score:.2f}")

    # 确定最终行业类型
    final_industry = keyword_industry if keyword_industry else 'generic'

    # 置信度不足时降级到generic
    if confidence < 0.6:
        classification_path.append(f"置信度{confidence:.2f}<0.6，降级到generic")
        return IndustryClassificationResult(
            industry_type='generic',
            display_name='综合行业',
            confidence=confidence,
            confidence_factors=confidence_factors,
            conservative_mode=True,
            classification_path=classification_path
        )

    return IndustryClassificationResult(
        industry_type=final_industry,
        display_name=get_display_name(final_industry),
        confidence=min(confidence, 1.0),
        confidence_factors=confidence_factors,
        conservative_mode=False,
        classification_path=classification_path
    )
```

```python
def classify_sub_industry(industry_type: str, company_info: dict,
                          metrics: dict) -> str:
    """
    在已识别的大行业内，进一步识别子行业

    参数:
    - industry_type: 大行业类型（如 'pharma'）
    - company_info: 公司信息
    - metrics: 财务指标

    返回: 子行业类型（如 'pharma_innovative'）或原行业类型
    """
    business_desc = company_info.get('business_description', '')
    company_name = company_info.get('name', '')

    # === 医药行业分化 ===
    if industry_type in ['pharma', 'pharma_mature', 'medical_device']:
        # 检查是否为创新药
        if is_innovative_pharma(metrics, business_desc):
            return 'pharma_innovative'

        # 检查是否为CXO
        cxo_keywords = ['CRO', 'CMO', 'CDMO', '医药外包', '药物研发服务']
        if any(kw in business_desc for kw in cxo_keywords):
            return 'pharma_cxo'

        # 检查是否为中药
        tcm_keywords = ['中药', '中成药', '中医药', '传统医药']
        if any(kw in business_desc + company_name for kw in tcm_keywords):
            return 'pharma_tcm'

    # === 科技行业分化 ===
    if industry_type in ['tech', 'software']:
        # 检查是否为SaaS
        saas_keywords = ['SaaS', '云服务', '订阅', 'ARR', '云计算']
        gross_margin = metrics.get('gross_margin', 0)
        if any(kw in business_desc for kw in saas_keywords) and gross_margin > 70:
            return 'tech_saas'
        else:
            return 'tech_traditional'

    # === 消费行业分化 ===
    if industry_type in ['consumer', 'brand_moat']:
        gross_margin = metrics.get('gross_margin', 0)
        if gross_margin > 70:
            return 'consumer_premium'
        elif gross_margin > 30:
            return 'consumer_mass'

    return industry_type  # 无法进一步分类，返回原类型

def is_innovative_pharma(metrics: dict, business_desc: str) -> bool:
    """
    判断是否为创新药研发公司

    条件（满足2个以上）:
    1. 研发费用率 > 30%
    2. 净利润为负或净利率 < 5%
    3. 业务描述包含创新药相关关键词
    4. 有临床管线描述
    """
    conditions = 0

    # 研发费用率高
    rd_ratio = metrics.get('rd_expense_ratio', 0)
    if rd_ratio > 30:
        conditions += 1

    # 亏损或微利
    net_margin = metrics.get('net_margin', 0)
    if net_margin < 5:
        conditions += 1

    # 关键词
    innovative_keywords = ['创新药', 'First-in-class', 'Best-in-class',
                           '临床试验', 'IND', 'NDA', '管线', 'Pipeline']
    if any(kw.lower() in business_desc.lower() for kw in innovative_keywords):
        conditions += 1

    return conditions >= 2
```

```python
def apply_brand_moat_valuation(metrics: dict, industry_config: dict) -> dict:
    """

    为什么用P/E而不是P/B：
    - P/B对轻资产品牌公司没有意义（账面价值不反映品牌价值）
    - 护城河公司的价值在于盈利能力的持续性，用P/E更直接
    """
    if not detect_brand_moat(metrics):
        return None

    # 更新行业配置
    industry_config['industry_type'] = 'brand_moat'
    industry_config['disable_methods'] = ['graham_number', 'pb']  # 也禁用P/B

    # 判断护城河等级
    moat_tier = classify_moat_tier(metrics)
    if moat_tier is None:
        moat_tier = 'moderate'

    # 获取P/E锚点范围
    pe_range = BRAND_MOAT_PE_ANCHORS[moat_tier]['pe_range']
    pe_low, pe_high = pe_range
    pe_mid = (pe_low + pe_high) / 2

    # 获取EPS
    eps = metrics.get('eps')
    if eps is None or eps <= 0:
        return {
            'method': 'pe_moat',
            'error': 'EPS为负或无效，护城河P/E估值不适用',
            'moat_tier': moat_tier
        }

    # 计算目标价区间
    target_low = eps * pe_low
    target_mid = eps * pe_mid
    target_high = eps * pe_high

    return {
        'method': 'pe_moat',
        'moat_tier': moat_tier,
        'pe_range': pe_range,
        'pe_applied': pe_mid,
        'target_price': target_mid,
        'target_range': (target_low, target_high),
        'eps_used': eps,
        'note': f'护城河等级: {moat_tier}，P/E锚点: {pe_low}-{pe_high}x'
    }
```

```python
def classify_distressed_type(company_info: dict, metrics: dict) -> str:
    """
    对困境企业进行分类

    返回: 'asset_intensive', 'contract_based', 'receivables_heavy', 或 'generic_distressed'
    """
    business_desc = company_info.get('business_description', '')
    company_name = company_info.get('name', '')
    combined = business_desc + company_name

    # 检查各类型关键词
    for category, config in DISTRESSED_CATEGORIES.items():
        if category == 'generic_distressed':
            continue
        keywords = config.get('keywords', [])
        if any(kw in combined for kw in keywords):
            return category

    # 检查财务特征
    receivables = metrics.get('accounts_receivable', 0)
    revenue = metrics.get('revenue', 1)
    if receivables / revenue > 0.5:  # 应收占营收比超过50%
        return 'receivables_heavy'

    return 'generic_distressed'

def distressed_valuation(metrics: dict, company_info: dict) -> dict:
    """
    困境企业估值框架（修正版）
    根据困境类型选择不同估值方法
    """
    results = {}

    # 1. 判断困境类型
    distressed_type = classify_distressed_type(company_info, metrics)
    config = DISTRESSED_CATEGORIES[distressed_type]
    results['distressed_type'] = distressed_type
    results['valuation_note'] = config['note']

    # 2. 根据类型选择估值方法
    if config['valuation_method'] == 'asset_replacement':
        # 有形资产密集型：资产重置价值
        fixed_assets = metrics.get('fixed_assets', 0)
        inventory = metrics.get('inventory', 0)
        # 假设固定资产折价50%，存货折价30%
        replacement_value = fixed_assets * 0.5 + inventory * 0.7
        results['asset_replacement'] = {
            'value': replacement_value / metrics.get('shares', 1),
            'fixed_assets': fixed_assets,
            'inventory': inventory,
            'note': '资产重置价值（固定资产50%折价 + 存货70%折价）'
        }
        # 补充EV/Sales作为参考
        if metrics.get('revenue', 0) > 0:
            ev = metrics['revenue'] * config['ev_sales_multiple']
            results['ev_sales_ref'] = {
                'value': ev / metrics.get('shares', 1),
                'multiple': config['ev_sales_multiple'],
                'note': '仅供参考'
            }

    elif config['valuation_method'] == 'backlog_value':
        # 合同/订单型：在手订单折现
        backlog = metrics.get('order_backlog', 0)  # 在手订单
        if backlog > 0:
            # 假设在手订单3年执行完毕，折现率10%，毛利率15%
            gross_margin = metrics.get('gross_margin', 15) / 100
            discount_rate = 0.10
            # 简化计算：订单价值 = 订单额 × 毛利率 × 折现因子
            backlog_pv = backlog * gross_margin * 0.85  # 3年平均折现约0.85
            results['backlog_value'] = {
                'value': backlog_pv / metrics.get('shares', 1),
                'order_backlog': backlog,
                'assumed_margin': gross_margin,
                'note': f'在手订单{backlog/1e8:.1f}亿，假设毛利率{gross_margin:.0%}，3年折现'
            }
        else:
            results['backlog_value'] = {
                'error': '无法获取在手订单数据',
                'fallback': 'ev_sales'
            }
            # 回退到EV/Sales
            if metrics.get('revenue', 0) > 0:
                results['ev_sales'] = {
                    'value': metrics['revenue'] * 0.4 / metrics.get('shares', 1),
                    'multiple': 0.4,  # 合同型用稍高倍数
                    'note': '回退到EV/Sales（无订单数据）'
                }

    elif config['valuation_method'] == 'receivables_recovery':
        # 应收账款型：回收率分析
        receivables = metrics.get('accounts_receivable', 0)
        # 假设困境企业应收回收率50%-70%
        recovery_low = receivables * 0.5
        recovery_high = receivables * 0.7
        results['receivables_recovery'] = {
            'value_range': (
                recovery_low / metrics.get('shares', 1),
                recovery_high / metrics.get('shares', 1)
            ),
            'receivables': receivables,
            'recovery_rate': '50%-70%',
            'note': f'应收账款{receivables/1e8:.1f}亿，预期回收率50%-70%'
        }

    else:
        # 通用困境：EV/Sales
        if metrics.get('revenue', 0) > 0:
            ev = metrics['revenue'] * config['ev_sales_multiple']
            results['ev_sales'] = {
                'value': ev / metrics.get('shares', 1),
                'multiple': config['ev_sales_multiple'],
                'note': config['note']
            }

    # 3. Net-Net分析（所有类型都计算）
    current_assets = metrics.get('current_assets', 0)
    total_liabilities = metrics.get('total_liabilities', 0)
    if current_assets > total_liabilities:
        net_net_value = (current_assets - total_liabilities) / metrics.get('shares', 1)
        results['net_net'] = {
            'value': net_net_value,
            'current_assets': current_assets,
            'total_liabilities': total_liabilities,
            'note': f'流动资产{current_assets/1e8:.1f}亿 > 总负债{total_liabilities/1e8:.1f}亿'
        }

    # 4. 退市风险评估
    delisting_risk = is_delisting_risk(metrics)
    if delisting_risk['level'] in ['MEDIUM', 'HIGH']:
        results['delisting_risk'] = delisting_risk

    return results
```

```python
            return result
    except Exception as e:
        logger.warning(f"计算EPS失败: {e}")

    return result

def validate_eps_consistency(eps_reported: float, eps_calculated: float,
                             threshold: float = 0.1) -> dict:
    """
    验证EPS一致性

    参数:
    - threshold: 允许的偏差阈值，默认10%

    返回:
    - is_consistent: 是否一致
    - deviation: 偏差百分比
    - recommendation: 建议使用哪个值
    """
    if eps_reported is None or eps_calculated is None:
        return {'is_consistent': True, 'deviation': None, 'recommendation': 'use_available'}

    if eps_reported == 0:
        return {'is_consistent': False, 'deviation': float('inf'), 'recommendation': 'use_calculated'}

    deviation = abs(eps_reported - eps_calculated) / abs(eps_reported)

    return {
        'is_consistent': deviation <= threshold,
        'deviation': deviation,
        'recommendation': 'use_reported' if deviation <= threshold else 'investigate'
    }
```

```python
def detect_data_contradictions(metrics: dict) -> list:
    """
    检测财务数据内部矛盾
    返回矛盾列表，每个矛盾包含类型、严重程度和详情
    """
    contradictions = []

    # 矛盾1：净利率与EPS不匹配
    net_margin = metrics.get('net_margin')
    eps = metrics.get('eps')
    revenue = metrics.get('revenue')
    shares = metrics.get('shares')

    if all(v is not None for v in [net_margin, eps, revenue, shares]) and shares > 0:
        implied_net_income = revenue * net_margin / 100  # 净利率隐含的净利润
        reported_net_income = eps * shares               # EPS隐含的净利润

        if implied_net_income > 0 and reported_net_income > 0:
            deviation = abs(implied_net_income - reported_net_income) / max(implied_net_income, 1)
            if deviation > 0.5:  # 偏差超过50%
                contradictions.append({
                    'type': 'net_margin_eps_mismatch',
                    'severity': 'high',
                    'detail': f"净利率{net_margin:.1f}%隐含净利{implied_net_income/1e8:.1f}亿，"
                             f"但EPS隐含净利{reported_net_income/1e8:.1f}亿，偏差{deviation:.0%}"
                })

    # 矛盾2：ROE历史跳变
    roe_current = metrics.get('roe')
    roe_5yr_avg = metrics.get('roe_5yr_avg')

    if roe_current is not None and roe_5yr_avg is not None:
        if abs(roe_current - roe_5yr_avg) > 20:  # ROE变化超过20个百分点
            contradictions.append({
                'type': 'roe_historical_jump',
                'severity': 'medium',
                'detail': f"当年ROE={roe_current:.1f}%与5年均值{roe_5yr_avg:.1f}%差异过大"
            })

    # 矛盾3：正净利但负OCF（连续两年）
    net_income = metrics.get('net_income')
    ocf = metrics.get('ocf')
    ocf_prev = metrics.get('ocf_prev_year')

    if net_income is not None and net_income > 0:
        if ocf is not None and ocf < 0:
            if ocf_prev is not None and ocf_prev < 0:
                contradictions.append({
                    'type': 'ni_ocf_divergence_persistent',
                    'severity': 'high',
                    'detail': f"净利润{net_income/1e8:.1f}亿为正，但经营现金流连续两年为负"
                             f"（今年{ocf/1e8:.1f}亿，去年{ocf_prev/1e8:.1f}亿）"
                })
            else:
                contradictions.append({
                    'type': 'ni_ocf_divergence',
                    'severity': 'medium',
                    'detail': f"净利润{net_income/1e8:.1f}亿为正，但经营现金流{ocf/1e8:.1f}亿为负"
                })

    # 矛盾4：EPS与报告EPS不一致
    eps_calculated = metrics.get('eps_calculated')
    eps_reported = metrics.get('eps_reported')

    if eps_calculated is not None and eps_reported is not None and eps_reported != 0:
        eps_deviation = abs(eps_calculated - eps_reported) / abs(eps_reported)
        if eps_deviation > 0.2:  # 偏差超过20%
            contradictions.append({
                'type': 'eps_inconsistency',
                'severity': 'low',
                'detail': f"计算EPS={eps_calculated:.2f}与报告EPS={eps_reported:.2f}不一致，偏差{eps_deviation:.0%}"
            })

    return contradictions

def get_data_confidence_score(contradictions: list) -> float:
    """
    基于矛盾计算数据可信度分数

    严重程度权重:
    - high: 扣0.3分
    - medium: 扣0.15分
    - low: 扣0.05分

    返回: 0.0-1.0的可信度分数
    """
    severity_weights = {'high': 0.3, 'medium': 0.15, 'low': 0.05}
    penalty = sum(severity_weights.get(c['severity'], 0.1) for c in contradictions)
    return max(0.0, 1.0 - penalty)

def generate_data_warning(contradictions: list, confidence: float) -> str:
    """
    当可信度<0.5时，生成报告开头的警告文本
    """
    if confidence >= 0.5:
        return None

    warning = f"""## ⚠️ 数据可信度警告

本报告使用的财务数据存在以下内部矛盾，可信度评分：**{confidence:.0%}**

| 矛盾类型 | 严重程度 | 详情 |
|:---------|:---------|:-----|
"""
    severity_map = {'high': '高', 'medium': '中', 'low': '低'}
    type_map = {
        'net_margin_eps_mismatch': '净利率/EPS不匹配',
        'roe_historical_jump': 'ROE历史跳变',
        'ni_ocf_divergence': '净利/OCF背离',
        'ni_ocf_divergence_persistent': '净利/OCF持续背离',
        'eps_inconsistency': 'EPS数据不一致'
    }

    for c in contradictions:
        type_name = type_map.get(c['type'], c['type'])
        severity = severity_map.get(c['severity'], c['severity'])
        warning += f"| {type_name} | {severity} | {c['detail']} |\n"

    warning += "\n**建议**：以下分析结论仅供参考，请核实原始财报数据后再做决策。\n"
    return warning
```
